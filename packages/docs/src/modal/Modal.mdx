## Default

<Playground>
  {() => {
    const [isShow, setIsShow] = useState(true)
    return (
      <>
        <Button onClick={() => setIsShow(!isShow)}>
          {isShow ? 'Show Modal' : 'Close Modal'}
        </Button>
        <Modal
          header={<div>Modal header</div>}
          body={<div>Modal body</div>}
          footer={<div>Modal footer</div>}
          isShow={isShow}
        />
      </>
    )
  }}
</Playground>
