<Playground>

    const { isOpen, onToggle } = useDisclosure();

    return (
      <ThemeProvider theme={LocalizeTheme}>
        <Button onClick={onToggle}>
          {isOpen ? 'Show Modal' : 'Close Modal'}
        </Button>
        <LocalizeModal isShow={isOpen} onClose={onToggle}>
          <div>LocalizeModal header</div>
          <div>LocalizeModal footer</div>
          <div>LocalizeModal body</div>
        </LocalizeModal>
      </ThemeProvider>
    )
  }}
