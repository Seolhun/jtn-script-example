# Alert

## Default

<Playground>
  {() => {
    const { isOpen, onToggle } = useDisclosure();
    return (
      <LocalizeThemeProvider>
        <LocalizeButton onClick={onToggle}>
          {isOpen ? 'Show Alert' : 'Close Alert'}
        </LocalizeButton>
        <LocalizeAlert position='top-center' onClick={onToggle} isShow={isOpen}>
          <div>Modal header</div>
          <div>Modal footer</div>
          <div>Modal body</div>
        </LocalizeAlert>
      </LocalizeThemeProvider>
    )
  }}
</Playground>
