## Default

<Playground>
  {() => {
    const [isChecked, setIsChecked] = useState(false)
    return (
      <ThemeProvider theme={LocalizeTheme}>

          onChange={(checked) => setIsChecked(!checked)}
        />
        <Switch
          htmlFor='example'
