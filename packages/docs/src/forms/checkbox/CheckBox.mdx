import React from 'react';

import { darkTheme } from '@seolhun/localize-components';
import { LocalizeThemeProvider } from '@seolhun/localize-components';

# CheckBox

## Default

<Playground>
  {() => {
    const [items, setItems] = React.useState([
      {
        label: 'Apple',
        value: 'Apple',
        checked: true,
      },
      {
        label: 'Peach',
        value: 'Peach',
        checked: false,
      },
    ])

    const handleOnChange = (selectedItem) => {
      const newItems = items.map((item) => {
        if (selectedItem.value === item.value) {
          return {
            ...selectedItem,
          };
        }
        return item;
      })
      setItems(newItems)
    }

    return (
      <LocalizeThemeProvider>
        {items.map((item) => (
          <LocalizeCheckBox
            key={item.label}
            item={item}
            onChange={handleOnChange}
            checked={item.checked}
          />
        ))}
      </LocalizeThemeProvider>
    )
  }}
</Playground>
