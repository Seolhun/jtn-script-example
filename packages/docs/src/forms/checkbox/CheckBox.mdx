import { useState, useCallback } from 'react';
import { Playground, Props } from 'docz';
import { CheckBox } from '@seolhun/localize-components-atomic';
import { ThemeProvider } from 'emotion-theming';
import { LocalizeTheme } from '@seolhun/localize-components';

# CheckBox

## Default

<Playground>
  {() => {
    const [items, setItems] = useState([
      {
        label: 'Apple',
        value: 'Apple',
        checked: true,
      },
      {
        label: 'Peach',
        value: 'Peach',
        checked: false,
      },
    ])

    const handleOnChange = useCallback((selectedItem) => {
      const newItems = items.map((item) => {
        if (selectedItem.value === item.value) {
          return {
            ...selectedItem,
          };
        }
        return item;
      })
      setItems(newItems)
    }, [items])

    return (
      <ThemeProvider theme={LocalizeTheme}>
        {items.map((item) => (
          <CheckBox
            key={item.label}
            item={item}
            onChange={handleOnChange}
            checked={item.checked}
          />
        ))}
      </ThemeProvider>
    )
  }}
</Playground>
