import { useState, useCallback } from 'react';
import { Playground, Props } from 'docz';
import { LocalizeCheckBox } from '@seolhun/localize-components-atomic';
import { LocalizeThemeProvider } from '@seolhun/localize-components';

# CheckBox

## Default

<Playground>
  {() => {
    const [items, setItems] = useState([
      {
        label: 'Apple',
        value: 'Apple',
        checked: true,
      },
      {
        label: 'Peach',
        value: 'Peach',
        checked: false,
      },
    ])

    const handleOnChange = useCallback((selectedItem) => {
      const newItems = items.map((item) => {
        if (selectedItem.value === item.value) {
          return {
            ...selectedItem,
          };
        }
        return item;
      })
      setItems(newItems)
    }, [items])

    return (
      <LocalizeThemeProvider>
        {items.map((item) => (
          <LocalizeCheckBox
            key={item.label}
            item={item}
            onChange={handleOnChange}
            checked={item.checked}
          />
        ))}
      </LocalizeThemeProvider>
    )
  }}
</Playground>
