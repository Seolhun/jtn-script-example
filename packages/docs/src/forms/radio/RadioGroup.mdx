import { useState, useCallback } from 'react';
import { Playground, Props } from 'docz';
import { RadioGroup, Radio } from '@seolhun/localize-components-atomic';
import { LocalizeThemeProvider } from '@seolhun/localize-components';

# RadioGroup

## Default

<Playground>
  {() => {
    const [checkedItem, setItem] = useState({
      label: 'label1',
      value: 'value1',
    })
    const handleOnChange = useCallback((item) => {
      setItem(item)
    }, [checkedItem])

    return (
      <LocalizeThemeProvider>
        <RadioGroup groupName='localize'>
          {props => (
            <>
              <Radio
                {...props}
                item={{
                  label: 'label1',
                  value: 'value1',
                }}
                checkedItem={checkedItem}
                onChange={handleOnChange}
                groupName='localize'
              />
              <Radio
                {...props}
                item={{
                  label: 'label2',
                  value: 'value2',
                }}
                checkedItem={checkedItem}
                onChange={handleOnChange}
                groupName='localize'
              />
            </>
          )}
        </RadioGroup>
      </LocalizeThemeProvider>
    )
  }}
</Playground>
