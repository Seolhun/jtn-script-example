import React from 'react';
import { Playground, Props } from 'docz';
import { LocalizeRadioGroup, LocalizeRadio } from '@seolhun/localize-components-atomic';

import { ThemeChangerProvider } from '../../ThemeChangerProvider'

# LocalizeRadioGroup

## Default

<Playground>
  {() => {
    const [checkedItem, setItem] = React.useState({
      label: 'label1',
      value: 'value1',
    })

    const handleOnChange = (item) => {
      setItem(item)
    }

    return (
      <ThemeChangerProvider>
        <LocalizeRadioGroup groupName='localize'>
          {props => (
            <>
              <LocalizeRadio
                {...props}
                item={{
                  label: 'label1',
                  value: 'value1',
                }}
                checkedItem={checkedItem}
                onChange={handleOnChange}
                groupName='localize'
              />
              <LocalizeRadio
                {...props}
                item={{
                  label: 'label2',
                  value: 'value2',
                }}
                checkedItem={checkedItem}
                onChange={handleOnChange}
                groupName='localize'
              />
            </>
          )}
        </LocalizeRadioGroup>
      </ThemeChangerProvider>
    )
  }}
</Playground>
