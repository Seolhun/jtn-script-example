import React from 'react';
import { Playground, Props } from 'docz';
import { LocalizeRadio } from '@seolhun/localize-components-atomic';

import { ThemeChangerProvider } from '../../ThemeChangerProvider'

# LocalizeRadio

## Used the checkedItem

<Playground>
  {() => {
    const [checkedItem, setItem] = React.useState({
      label: 'label1',
      value: 'value1',
    })

    const handleOnChange = (item) => {
      setItem(item)
    }

    return (
      <ThemeChangerProvider>
        <LocalizeRadio
          item={{
            label: 'label1',
            value: 'value1',
          }}
          checkedItem={checkedItem}
          onChange={handleOnChange}
          groupName='localize'
        />
        <LocalizeRadio
          item={{
            label: 'label2',
            value: 'value2',
          }}
          checkedItem={checkedItem}
          onChange={handleOnChange}
          groupName='localize'
        />
      </ThemeChangerProvider>
    )
  }}
</Playground>
